const map = L.map('map', { zoomControl: false, preferCanvas: true }).setView([41.1579, -8.6291], 13); L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', { attribution: '© CARTO', updateWhenZooming: false, updateWhenIdle: true }).addTo(map); let busLayer = L.layerGroup().addTo(map); function getBusColor(line) { if (!line) return '#ffffff'; const s = line.toString().toUpperCase().trim(); if (s.endsWith('M')) return '#000000'; if (s.startsWith('5')) return '#fcd116'; if (s.startsWith('2') || s.startsWith('3') || s.startsWith('4') || s.includes('ZC')) return '#187ec2'; if (s.startsWith('6')) return '#00ac00'; if (s.startsWith('7')) return '#ff0000'; if (s.startsWith('8')) return '#a347ff'; if (s.startsWith('9')) return '#ff7900'; return '#ffffff'; } async function updateBuses() { // FIX 2: Do not run this function if the user is zooming or dragging the map. // This stops the visual bugs entirely. if (map.dragging.moving() || map._animatingZoom) return; try { // FIX 3: Added cache: 'no-store' to guarantee the browser forces a real network request const response = await fetch('https://broker.fiware.urbanplatform.portodigital.pt/v2/entities?q=vehicleType==bus&limit=1000&t=' + Date.now(), { cache: 'no-store' }); const data = await response.json(); document.getElementById('counter').innerText = `Buses: ${data.length}`; const newMarkers = []; data.forEach(bus => { const fullId = bus.id; const idParts = fullId.split(':'); const fleetId = idParts[idParts.length - 1]; let line = bus.name ? bus.name.value.replace('STCP', '').replace(fleetId, '').trim() : "??"; if (bus.location && bus.location.value && bus.location.value.coordinates) { const [lon, lat] = bus.location.value.coordinates; const marker = L.circleMarker([lat, lon], { radius: 9, fillColor: getBusColor(line), color: "#ffffff", weight: 2, opacity: 1, fillOpacity: 1, interactive: true }); marker.bindPopup(`${line} | ${fleetId}`); newMarkers.push(marker); } }); // FIX 1: Instead of clearing the map and drawing one by one, // we bundle all the new dots together and swap the layers instantly. const newLayer = L.layerGroup(newMarkers); map.removeLayer(busLayer); busLayer = newLayer; map.addLayer(busLayer); } catch (e) { console.error("Update failed:", e); } } updateBuses(); // FIX 3 (continued): Slowed down to 3 seconds. The buses don't move fast enough to justify 0.5s anyway, // and this prevents the API from blocking you. setInterval(updateBuses, 3000);
